int minimum_cost_calculator(int A[1000],int n,int c1,int c2,int c3)
{
    int min1=0,p;
    for(int i=0;i<n;i++)
    {

        min1=min1+smaller(c2,p);
    }
     return smaller(min1,c3);
}
